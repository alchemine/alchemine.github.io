# 1. Probability Density Function
{% highlight python linenos %}
import numpy as np
import matplotlib.pyplot as plt
from scipy.stats import norm

X = np.arange(-10, 10, 0.001)
plt.plot(X, norm.pdf(X, 0, 2));

## Code
{% highlight python linenos %}
def solution(n, t, m, p):
    def convertor(val, n):
        rst = ''
        while True:
            val, r = val//n, dic[val%n]
            rst = f"{r}{rst}"
            if val < n:
                return f"{dic[val]}{rst}" if val > 0 else rst

    dic = list('0123456789ABCDEF')
    ser = ""
    rst = ""

    val = 0
    idx = p-1
    for _ in range(t):
        while len(ser) < idx+1:
            ser = f"{ser}{convertor(val, n)}"
            val += 1
        rst = f"{rst}{ser[idx]}"
        idx += m
    return rst
{% endhighlight %}

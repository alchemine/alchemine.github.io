## Code
{% highlight python linenos %}
class Solution:
    def checkPossibility(self, nums: List[int]) -> bool:
        if len(nums) < 3:
            return True

        chance = 1
        for i in range(len(nums)-2):
            a, b, c = nums[i:i+3]

            if (a <= b <= c):
                continue
            elif (a > b > c):
                return False
            else:
                if chance == 0:
                    return False
                chance -= 1
                if (a <= b) and (b > c):
                    if a <= c:
                        nums[i+1] = nums[i+2]
                    else:
                        nums[i+2] = nums[i+1]
                elif (a > b) and (b <= c):
                    if a <= c:
                        nums[i+1] = nums[i+2]
                    else:
                        nums[i] = nums[i+1]
        return True
{% endhighlight %}


## Complexity
$O(|\text{nums}|)$
